* Instalação
  
  Crie este projeto com maven

  Rode esse comando na sua máquina para criar o seu projeto
  Um diretório "MeuPrimeiroProjetoCucumber" será criado no mesmo
  diretório em que vc deu esse comando. 
  
   #+NAME: mvn archetype:generate   
   #+BEGIN_SRC shell :session s1 :results output :exports both
      #starts code
      mvn archetype:generate                      \
      "-DarchetypeGroupId=io.cucumber"           \
      "-DarchetypeArtifactId=cucumber-archetype" \
      "-DarchetypeVersion=4.2.6.1"               \
      "-DgroupId=br.eteczl.tqs"                  \
      "-DartifactId=MeuPrimeiroProjetoCucumber"               \
      "-Dpackage=br.eteczl.tqs"                  \
      "-Dversion=1.0.0-SNAPSHOT"                 \
      "-DinteractiveMode=false"
   #+END_SRC
