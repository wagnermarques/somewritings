#+begin_src sqlite :db /home/wagner/mnt/pen/fzlbpms/data/orgmode-sqlite-dbs/ :var orgtable=ctaspgar :colnames yes

  drop table if exists ctaspagar;
  create table ctaspagar(conta text, vcto date, vlr real);
